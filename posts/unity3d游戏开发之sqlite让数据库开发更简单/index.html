什么是 SQLite？

SQLite是一款轻型的数据库，是遵守 ACID 的关系型数据库管理系统，它包含在一个相对小的 C 库中，以嵌入式作为它的设计目标，它占用资源非常的低，因此适合在嵌入式设备如 Android、Ruby on Rails 等中使用。它能够支持 Windows/Linux/Unix 等等主流的操作系统，同时能够跟和 C、C++、Ruby、Python、C#、PHP、Java 等编程语言相结合。SQLite 是一个以文件形式存在的关系型数据库，尽管无法实现分布式和横向扩展，可是作为一个轻量级的嵌入式数据库，它不需要系统提供服务支持，通过 SDK 直接操作文件避免了对数据库维护的相关事务，从这个角度来讲它是一个出色的数据库。

为什么要选择 SQLite

好了，在了解了 SQLite 后，我们来了解下 SQLite 有哪些让我们心动的特性，或者说我们为什么要选择 SQLite，因为在这个世界上我们有太多的数据库可以选择，诸如 Oracle、MySQL、SQLServer、DB2、NoSQL、MongoDB 等等：

  • ACID 事务
  • 零配置 – 无需安装和管理配置
  • 储存在单一磁盘文件中的一个完整的数据库
  • 数据库文件可以在不同字节顺序的机器间自由的共享
  • 支持数据库大小至 2TB
  • 足够小, 大致 13 万行 C 代码, 4.43M
  • 比一些流行的数据库在大部分普通数据库操作要快—SQLite 读写效率如此之高，会使用其他数据库的理由是？
  • 简单, 轻松的 API
  • 包含 TCL 绑定, 同时通过 Wrapper 支持其他语言的绑定
  • 良好注释的源代码, 并且有着90%以上的测试覆盖率
  • 独立: 没有额外依赖
  • 源码完全的开源, 你可以用于任何用途, 包括出售它
  • 支持多种开发语言，C, C++, PHP, Perl, Java, C#,Python, Ruby 等

Unity3D 中的 SQLite

在 Unity3D 中使用 SQLite，我们首先要明白这样一件事情，即我们这里的使用的 SQLite 并非是通常意义上的 SQLite.NET,而是经过移植后的 Mono.Data.Sqlite。因为 Unity3D 基于 Mono，因此使用移植后的 Mono.Data.Sqlite 能够减少我们的项目在不同平台上出现各种各样的问题。在 Unity3D 中使用的 SQLite 以 Mono.Data.Sqlite.dll 即动态链接库的形式给出，因此我们需要将这个文件放置在项目目录下的 Plugins 文件夹中，此外我们需要 System.Data.dll 或者 Mono.Data.dll 这两个文件添加到 Plugins 目录中，因为我们需要的部分数据相关的 API 或者类都定义在这两个文件当中，这些文件可以从这里直接下载。

PS：博主注意到在网上有使用 Mono.Data.SQLiteClient.dll 这个库实现在 Unity3D 操作 SQLite 数据库的相关文章，博主大概看了下，感觉和使用 Mono.Data.Sqlite.dll 这个库大同小异，大家喜欢哪个就用哪个吧！哈哈！博主在开源社区找到一个版本库，据说可以同时支持.NET 和 Mono，如果大家感兴趣欢迎大家去测试啊，哈哈!

在正式开始写代码前，我们首先来回顾下通常情况下数据库读写的基本流程吧！

  • 定义数据库连接字符串(ConnectionString)完成数据库连接的构造，建立或者打开一个数据库。
  • 定义相关的 SQL 命令(Command)通过这些命令实现对数据库的增加、删除、更新、读取四种基本功能。
  • 在完成各种数据库操作后及时关闭数据库连接，解除对数据库的连接和引用。

SQLite 作为一款优秀的数据库，在为其编写数据库相关代码时同样遵循这样的流程，考虑到对数据库的增加、删除、更新、读取四种操作具有类似性和统一性，因此在动手写 Unity3D 脚本前，首先让我们来编写一个 SQLite 的辅助类 SQLiteHelper.cs。该类代码定义如下：

  1using UnityEngine;
  2using System.Collections;
  3using Mono.Data.Sqlite;
  4using System;
  5
  6public class SQLiteHelper
  7{
  8	/// <summary>
  9	/// 数据库连接定义
 10	/// </summary>
 11	private SqliteConnection dbConnection;
 12
 13	/// <summary>
 14	/// SQL命令定义
 15	/// </summary>
 16	private SqliteCommand dbCommand;
 17
 18	/// <summary>
 19	/// 数据读取定义
 20	/// </summary>
 21	private SqliteDataReader dataReader;
 22
 23	/// <summary>
 24	/// 构造函数	
 25	/// </summary>
 26	/// <param name="connectionString">数据库连接字符串</param>
 27	public SQLiteHelper(string connectionString)
 28	{
 29		try{
 30			//构造数据库连接
 31			dbConnection=new SqliteConnection(connectionString);
 32			//打开数据库
 33			dbConnection.Open();
 34		}catch(Exception e)
 35		{
 36			Debug.Log(e.Message);
 37		}
 38	}
 39
 40	/// <summary>
 41	/// 执行SQL命令
 42	/// </summary>
 43	/// <returns>The query.</returns>
 44	/// <param name="queryString">SQL命令字符串</param>
 45	public SqliteDataReader ExecuteQuery(string queryString)
 46	{
 47		dbCommand = dbConnection.CreateCommand();
 48		dbCommand.CommandText = queryString;
 49		dataReader = dbCommand.ExecuteReader();
 50		return dataReader;
 51	}
 52
 53	/// <summary>
 54	/// 关闭数据库连接
 55	/// </summary>
 56	public void CloseConnection()
 57	{
 58		//销毁Command
 59		if(dbCommand != null){
 60			dbCommand.Cancel();
 61		}
 62		dbCommand = null;
 63
 64		//销毁Reader
 65		if(dataReader != null){
 66			dataReader.Close();
 67		}
 68		dataReader = null;
 69
 70		//销毁Connection
 71		if(dbConnection != null){
 72			dbConnection.Close();
 73		}
 74		dbConnection = null;
 75	}
 76
 77	/// <summary>
 78	/// 读取整张数据表
 79	/// </summary>
 80	/// <returns>The full table.</returns>
 81	/// <param name="tableName">数据表名称</param>
 82	public SqliteDataReader ReadFullTable(string tableName)
 83	{
 84		string queryString = "SELECT * FROM " + tableName;
 85		return ExecuteQuery (queryString);
 86	}
 87
 88	/// <summary>
 89	/// 向指定数据表中插入数据
 90	/// </summary>
 91	/// <returns>The values.</returns>
 92	/// <param name="tableName">数据表名称</param>
 93	/// <param name="values">插入的数值</param>
 94	public SqliteDataReader InsertValues(string tableName,string[] values)
 95	{
 96		//获取数据表中字段数目
 97		int fieldCount=ReadFullTable(tableName).FieldCount;
 98		//当插入的数据长度不等于字段数目时引发异常
 99		if(values.Length!=fieldCount){
100			throw new SqliteException("values.Length!=fieldCount");
101		}
102
103		string queryString = "INSERT INTO " + tableName + " VALUES (" + values[0];
104		for(int i=1; i<values.Length; i++)
105		{
106			queryString+=", " + values[i];
107		}
108		queryString += " )";
109		return ExecuteQuery(queryString);
110	}
111
112	/// <summary>
113	/// 更新指定数据表内的数据
114	/// </summary>
115	/// <returns>The values.</returns>
116	/// <param name="tableName">数据表名称</param>
117	/// <param name="colNames">字段名</param>
118	/// <param name="colValues">字段名对应的数据</param>
119	/// <param name="key">关键字</param>
120	/// <param name="value">关键字对应的值</param>
121	public SqliteDataReader UpdateValues(string tableName,string[] colNames,string[] colValues,string key,string operation,string value)
122	{
123		//当字段名称和字段数值不对应时引发异常
124		if(colNames.Length!=colValues.Length) {
125			throw new SqliteException("colNames.Length!=colValues.Length");
126		}
127
128		string queryString = "UPDATE " + tableName + " SET " + colNames[0] + "=" + colValues[0];
129		for(int i=1; i<colValues.Length; i++) 
130		{
131			queryString+=", " + colNames[i] + "=" + colValues[i];
132		}
133		queryString += " WHERE " + key + operation + value;
134		return ExecuteQuery(queryString);
135	}
136
137	/// <summary>
138	/// 删除指定数据表内的数据
139	/// </summary>
140	/// <returns>The values.</returns>
141	/// <param name="tableName">数据表名称</param>
142	/// <param name="colNames">字段名</param>
143	/// <param name="colValues">字段名对应的数据</param>
144	public SqliteDataReader DeleteValuesOR(string tableName,string[] colNames,string[] operations,string[] colValues)
145	{
146		//当字段名称和字段数值不对应时引发异常
147		if(colNames.Length!=colValues.Length || operations.Length!=colNames.Length || operations.Length!=colValues.Length) {
148			throw new SqliteException("colNames.Length!=colValues.Length || operations.Length!=colNames.Length || operations.Length!=colValues.Length");
149		}
150
151		string queryString = "DELETE FROM " + tableName + " WHERE " + colNames[0] + operations[0] + colValues[0];
152		for(int i=1; i<colValues.Length; i++) 
153		{
154			queryString+="OR " + colNames[i] + operations[0] + colValues[i];
155		}
156		return ExecuteQuery(queryString);
157	}
158
159	/// <summary>
160	/// 删除指定数据表内的数据
161	/// </summary>
162	/// <returns>The values.</returns>
163	/// <param name="tableName">数据表名称</param>
164	/// <param name="colNames">字段名</param>
165	/// <param name="colValues">字段名对应的数据</param>
166	public SqliteDataReader DeleteValuesAND(string tableName,string[] colNames,string[] operations,string[] colValues)
167	{
168		//当字段名称和字段数值不对应时引发异常
169		if(colNames.Length!=colValues.Length || operations.Length!=colNames.Length || operations.Length!=colValues.Length) {
170			throw new SqliteException("colNames.Length!=colValues.Length || operations.Length!=colNames.Length || operations.Length!=colValues.Length");
171		}
172
173		string queryString = "DELETE FROM " + tableName + " WHERE " + colNames[0] + operations[0] + colValues[0];
174		for(int i=1; i<colValues.Length; i++) 
175		{
176			queryString+=" AND " + colNames[i] + operations[i] + colValues[i];
177		}
178		return ExecuteQuery(queryString);
179	}
180
181	/// <summary>
182	/// 创建数据表
183	/// </summary> +
184	/// <returns>The table.</returns>
185	/// <param name="tableName">数据表名</param>
186	/// <param name="colNames">字段名</param>
187	/// <param name="colTypes">字段名类型</param>
188	public SqliteDataReader CreateTable(string tableName,string[] colNames,string[] colTypes)
189	{
190		string queryString = "CREATE TABLE " + tableName + "( " + colNames [0] + " " + colTypes [0];
191		for (int i=1; i<colNames.Length; i++) 
192		{
193			queryString+=", " + colNames[i] + " " + colTypes[i];
194		}
195		queryString+= "  ) ";
196		return ExecuteQuery(queryString);
197	}
198
199	/// <summary>
200	/// Reads the table.
201	/// </summary>
202	/// <returns>The table.</returns>
203	/// <param name="tableName">Table name.</param>
204	/// <param name="items">Items.</param>
205	/// <param name="colNames">Col names.</param>
206	/// <param name="operations">Operations.</param>
207	/// <param name="colValues">Col values.</param>
208	public SqliteDataReader ReadTable(string tableName,string[] items,string[] colNames,string[] operations, string[] colValues)
209	{
210		string queryString = "SELECT " + items [0];
211		for (int i=1; i<items.Length; i++) 
212		{
213			queryString+=", " + items[i];
214		}
215		queryString += " FROM " + tableName + " WHERE " + colNames[0] + " " +  operations[0] + " " + colValues[0];
216		for (int i=0; i<colNames.Length; i++) 
217		{
218			queryString+=" AND " + colNames[i] + " " + operations[i] + " " + colValues[0] + " ";
219		}
220		return ExecuteQuery(queryString);
221	}
222}

SQLiteHelper 类主要实现了数据库、数据表的创建以及数据表中记录的增加、删除、更新、读取四种基本功能。该类最初由国外的 Unity3D 开发者发布在Unity3D 官方论坛,后来经宣雨松使用C#进行重写，我在此基础上进行了完善，再此对两位大神的无私付出表示感谢。这里要说明的有三点：

  • 一、在 Unity3D 编辑器下生成数据库文件(.db)默认位于和 Assets 目录同级的位置，即项目的工程文件夹中。我们可以通过修改路径在改变数据库文件的存储位置，具体来讲： Windows 平台：data source=Application.dataPath/数据库名称.db IOS 平台：data source=Application.persistentDataPath/数据库名称.db Android 平台：URL=file:Application.persistentDataPath/数据库名称.db(我想说 Android 平台就是个奇葩，搞什么特殊化嘛)

  • 二、确保 Unity3D 编辑器中的.NET 版本和 MonoDevelop 中的.NET 版本都为 2.0 版本，在 Unity3D 中打包导出的程序可能不会保留数据库文件，因此需要手动将数据库文件拷贝到相应的位置，当然更加合理的方案是将数据库文件存放到 StreamingAssets 文件夹下，然后在第一次加载游戏的时候将数据库文件复制到对应平台上的存放位置。

  • 三、在使用 InsertValues 方法时请参考 SQLite 中字段类型与 C#中数据类型的对应关系，博主目前测试了 int 类型和 string 类型都没有什么问题，更多类型的数据请大家自行测试然后告诉博主测试的结果，如果大家有兴趣扩展这个辅助类的话可以自行去扩展哦，嘿嘿！

好了，千呼万唤始出来的时候到了，下面我们以一个实例来完成今天的项目讲解，因为我们已经定义好了 SQLite 的辅助类，因此我们可以快速地编写出下面的脚本代码：

 1using UnityEngine;
 2using System.Collections;
 3using System.IO;
 4using Mono.Data.Sqlite;
 5
 6public class SQLiteDemo : MonoBehaviour 
 7{
 8	/// <summary>
 9	/// SQLite数据库辅助类
10	/// </summary>
11	private SQLiteHelper sql;
12
13	void Start () 
14	{
15		//创建名为sqlite4unity的数据库
16		sql = new SQLiteHelper("data source=sqlite4unity.db");
17
18		//创建名为table1的数据表
19		sql.CreateTable("table1",new string[]{"ID","Name","Age","Email"},new string[]{"INTEGER","TEXT","INTEGER","TEXT"});
20
21		//插入两条数据
22		sql.InsertValues("table1",new string[]{"'1'","'张三'","'22'","'Zhang3@163.com'"});
23		sql.InsertValues("table1",new string[]{"'2'","'李四'","'25'","'Li4@163.com'"});
24
25		//更新数据，将Name="张三"的记录中的Name改为"Zhang3"
26		sql.UpdateValues("table1", new string[]{"Name"}, new string[]{"'Zhang3'"}, "Name", "=", "'张三'");
27
28		//插入3条数据
29		sql.InsertValues("table1",new string[]{"3","'王五'","25","'Wang5@163.com'"});
30		sql.InsertValues("table1",new string[]{"4","'王五'","26","'Wang5@163.com'"});
31		sql.InsertValues("table1",new string[]{"5","'王五'","27","'Wang5@163.com'"});
32
33		//删除Name="王五"且Age=26的记录,DeleteValuesOR方法类似
34		sql.DeleteValuesAND("table1", new string[]{"Name","Age"}, new string[]{"=","="}, new string[]{"'王五'","'26'"});
35
36		//读取整张表
37		SqliteDataReader reader = sql.ReadFullTable ("table1");
38		while(reader.Read()) 
39		{
40			//读取ID
41			Debug.Log(reader.GetInt32(reader.GetOrdinal("ID")));
42			//读取Name
43			Debug.Log(reader.GetString(reader.GetOrdinal("Name")));
44			//读取Age
45			Debug.Log(reader.GetInt32(reader.GetOrdinal("Age")));
46			//读取Email
47			Debug.Log(reader.GetString(reader.GetOrdinal("Email")));
48		}
49
50		//读取数据表中Age>=25的所有记录的ID和Name
51		reader = sql.ReadTable ("table1", new string[]{"ID","Name"}, new string[]{"Age"}, new string[]{">="}, new string[]{"'25'"});
52		while(reader.Read()) 
53		{
54			//读取ID
55			Debug.Log(reader.GetInt32(reader.GetOrdinal("ID")));
56			//读取Name
57			Debug.Log(reader.GetString(reader.GetOrdinal("Name")));
58		}
59
60		//自定义SQL,删除数据表中所有Name="王五"的记录
61		sql.ExecuteQuery("DELETE FROM table1 WHERE NAME='王五'");
62
63		//关闭数据库连接
64		sql.CloseConnection();
65	}
66}
